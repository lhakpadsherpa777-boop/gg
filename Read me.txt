







import React, { useState, useEffect } from 'react';
import { Heart, Droplet, Moon, Activity, Apple, Brain, CheckCircle, Calendar, TrendingUp, BarChart3 } from 'lucide-react';

export default function WellnessChecker() {
  const [answers, setAnswers] = useState({
    sleep: '',
    water: '',
    exercise: '',
    meals: '',
    stress: '',
    screenTime: ''
  });
  
  const [showResults, setShowResults] = useState(false);
  const [showHistory, setShowHistory] = useState(false);
  const [history, setHistory] = useState([]);

  useEffect(() => {
    const loadHistory = async () => {
      try {
        const result = await window.storage.get('wellness-history');
        if (result && result.value) {
          setHistory(JSON.parse(result.value));
        }
      } catch (error) {
        console.log('No history found yet');
      }
    };
    loadHistory();
  }, []);

  const questions = [
    {
      id: 'sleep',
      question: 'How many hours did you sleep last night?',
      icon: Moon,
      options: [
        { value: 'less5', label: 'Less than 5 hours', score: 0 },
        { value: '5to6', label: '5-6 hours', score: 1 },
        { value: '7to9', label: '7-9 hours', score: 3 },
        { value: 'more9', label: 'More than 9 hours', score: 2 }
      ]
    },
    {
      id: 'water',
      question: 'How much water did you drink today?',
      icon: Droplet,
      options: [
        { value: 'less4', label: 'Less than 4 glasses', score: 0 },
        { value: '4to6', label: '4-6 glasses', score: 2 },
        { value: '7to8', label: '7-8 glasses', score: 3 },
        { value: 'more8', label: 'More than 8 glasses', score: 3 }
      ]
    },
    {
      id: 'exercise',
      question: 'Did you get physical activity today?',
      icon: Activity,
      options: [
        { value: 'none', label: 'No activity', score: 0 },
        { value: 'light', label: 'Light activity (10-20 min)', score: 2 },
        { value: 'moderate', label: 'Moderate activity (30+ min)', score: 3 },
        { value: 'intense', label: 'Intense workout', score: 3 }
      ]
    },
    {
      id: 'meals',
      question: 'How would you rate your meals today?',
      icon: Apple,
      options: [
        { value: 'poor', label: 'Mostly processed/fast food', score: 0 },
        { value: 'fair', label: 'Mixed healthy and unhealthy', score: 2 },
        { value: 'good', label: 'Mostly balanced meals', score: 3 },
        { value: 'excellent', label: 'All nutritious, balanced meals', score: 3 }
      ]
    },
    {
      id: 'stress',
      question: 'How did you manage stress today?',
      icon: Brain,
      options: [
        { value: 'high', label: 'Very stressed, no management', score: 0 },
        { value: 'moderate', label: 'Some stress, minimal breaks', score: 1 },
        { value: 'managed', label: 'Took breaks, practiced mindfulness', score: 3 },
        { value: 'low', label: 'Felt calm and balanced', score: 3 }
      ]
    },
    {
      id: 'screenTime',
      question: 'How much recreational screen time today?',
      icon: Activity,
      options: [
        { value: 'excessive', label: 'More than 5 hours', score: 0 },
        { value: 'high', label: '3-5 hours', score: 1 },
        { value: 'moderate', label: '1-3 hours', score: 2 },
        { value: 'low', label: 'Less than 1 hour', score: 3 }
      ]
    }
  ];

  const handleAnswer = (questionId, value) => {
    setAnswers(prev => ({ ...prev, [questionId]: value }));
  };

  const calculateScore = () => {
    let totalScore = 0;
    questions.forEach(q => {
      const selectedOption = q.options.find(opt => opt.value === answers[q.id]);
      if (selectedOption) totalScore += selectedOption.score;
    });
    return totalScore;
  };

  const getScoreDetails = (score) => {
    const maxScore = 18;
    const percentage = (score / maxScore) * 100;
    
    if (percentage >= 85) {
      return {
        level: 'Excellent',
        color: 'text-green-600',
        bgColor: 'bg-green-50',
        borderColor: 'border-green-200',
        message: 'Outstanding! You\'re maintaining excellent wellness habits. Keep up the great work!',
        tips: []
      };
    } else if (percentage >= 70) {
      return {
        level: 'Good',
        color: 'text-blue-600',
        bgColor: 'bg-blue-50',
        borderColor: 'border-blue-200',
        message: 'Great job! You\'re doing well with your daily wellness routine.',
        tips: getTips(score)
      };
    } else if (percentage >= 50) {
      return {
        level: 'Fair',
        color: 'text-yellow-600',
        bgColor: 'bg-yellow-50',
        borderColor: 'border-yellow-200',
        message: 'You\'re on the right track, but there\'s room for improvement.',
        tips: getTips(score)
      };
    } else {
      return {
        level: 'Needs Improvement',
        color: 'text-red-600',
        bgColor: 'bg-red-50',
        borderColor: 'border-red-200',
        message: 'Your wellness routine could use some attention. Small changes can make a big difference!',
        tips: getTips(score)
      };
    }
  };

  const getTips = (score) => {
    const tips = [];
    questions.forEach(q => {
      const selectedOption = q.options.find(opt => opt.value === answers[q.id]);
      if (selectedOption && selectedOption.score < 3) {
        const tipMap = {
          sleep: 'Try to get 7-9 hours of sleep by maintaining a consistent bedtime',
          water: 'Aim for at least 8 glasses of water throughout the day',
          exercise: 'Include at least 30 minutes of physical activity daily',
          meals: 'Focus on balanced meals with vegetables, proteins, and whole grains',
          stress: 'Practice stress management techniques like deep breathing or meditation',
          screenTime: 'Reduce recreational screen time and take regular breaks'
        };
        tips.push(tipMap[q.id]);
      }
    });
    return tips;
  };

  const saveToHistory = async (score, level) => {
    const newEntry = {
      date: new Date().toISOString(),
      score: score,
      level: level,
      answers: { ...answers }
    };
    
    const updatedHistory = [newEntry, ...history].slice(0, 30); // Keep last 30 entries
    setHistory(updatedHistory);
    
    try {
      await window.storage.set('wellness-history', JSON.stringify(updatedHistory));
    } catch (error) {
      console.error('Error saving history:', error);
    }
  };

  const handleSubmit = () => {
    const allAnswered = questions.every(q => answers[q.id] !== '');
    if (allAnswered) {
      const score = calculateScore();
      const details = getScoreDetails(score);
      saveToHistory(score, details.level);
      setShowResults(true);
    } else {
      alert('Please answer all questions before checking your wellness score!');
    }
  };

  const handleReset = () => {
    setAnswers({
      sleep: '',
      water: '',
      exercise: '',
      meals: '',
      stress: '',
      screenTime: ''
    });
    setShowResults(false);
  };

  const formatDate = (dateString) => {
    const date = new Date(dateString);
    const today = new Date();
    const yesterday = new Date(today);
    yesterday.setDate(yesterday.getDate() - 1);
    
    if (date.toDateString() === today.toDateString()) {
      return 'Today';
    } else if (date.toDateString() === yesterday.toDateString()) {
      return 'Yesterday';
    } else {
      return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
    }
  };

  const getAverageScore = () => {
    if (history.length === 0) return 0;
    const sum = history.reduce((acc, entry) => acc + entry.score, 0);
    return (sum / history.length).toFixed(1);
  };

  const getTrend = () => {
    if (history.length < 2) return 'neutral';
    const recent = history.slice(0, 3).reduce((acc, e) => acc + e.score, 0) / Math.min(3, history.length);
    const older = history.slice(3, 6).reduce((acc, e) => acc + e.score, 0) / Math.max(1, history.slice(3, 6).length);
    
    if (recent > older + 1) return 'up';
    if (recent < older - 1) return 'down';
    return 'neutral';
  };

  if (showHistory) {
    const avgScore = getAverageScore();
    const trend = getTrend();
    
    return (
      <div className="min-h-screen bg-gradient-to-br from-purple-50 to-blue-50 p-6">
        <div className="max-w-2xl mx-auto">
          <div className="bg-white rounded-2xl shadow-xl p-8">
            <div className="text-center mb-8">
              <BarChart3 className="w-16 h-16 mx-auto mb-4 text-purple-600" />
              <h1 className="text-3xl font-bold text-gray-800 mb-2">Your Wellness History</h1>
              <p className="text-gray-600">Track your progress over time</p>
            </div>

            {history.length > 0 && (
              <div className="grid grid-cols-2 gap-4 mb-6">
                <div className="bg-purple-50 rounded-xl p-4 text-center">
                  <div className="text-2xl font-bold text-purple-600">{avgScore}/18</div>
                  <div className="text-sm text-gray-600">Average Score</div>
                </div>
                <div className="bg-blue-50 rounded-xl p-4 text-center">
                  <div className="flex items-center justify-center">
                    {trend === 'up' && <TrendingUp className="w-6 h-6 text-green-600 mr-2" />}
                    {trend === 'down' && <TrendingUp className="w-6 h-6 text-red-600 mr-2 rotate-180" />}
                    {trend === 'neutral' && <Activity className="w-6 h-6 text-gray-600 mr-2" />}
                    <span className="text-2xl font-bold text-gray-800">{history.length}</span>
                  </div>
                  <div className="text-sm text-gray-600">Check-ins</div>
                </div>
              </div>
            )}

            <div className="space-y-3 mb-6 max-h-96 overflow-y-auto">
              {history.length === 0 ? (
                <div className="text-center py-12 text-gray-500">
                  <Calendar className="w-12 h-12 mx-auto mb-3 opacity-50" />
                  <p>No history yet. Complete your first wellness check!</p>
                </div>
              ) : (
                history.map((entry, index) => {
                  const details = getScoreDetails(entry.score);
                  return (
                    <div key={index} className={`${details.bgColor} ${details.borderColor} border rounded-xl p-4`}>
                      <div className="flex items-center justify-between">
                        <div>
                          <div className="flex items-center">
                            <Calendar className="w-4 h-4 mr-2 text-gray-600" />
                            <span className="font-semibold text-gray-800">{formatDate(entry.date)}</span>
                          </div>
                          <div className={`text-sm ${details.color} font-medium mt-1`}>{entry.level}</div>
                        </div>
                        <div className={`text-3xl font-bold ${details.color}`}>
                          {entry.score}/18
                        </div>
                      </div>
                    </div>
                  );
                })
              )}
            </div>

            <button
              onClick={() => setShowHistory(false)}
              className="w-full bg-purple-600 hover:bg-purple-700 text-white font-semibold py-3 px-6 rounded-lg transition"
            >
              Back to Wellness Check
            </button>
          </div>
        </div>
      </div>
    );
  }

  if (showResults) {
    const score = calculateScore();
    const details = getScoreDetails(score);
    
    return (
      <div className="min-h-screen bg-gradient-to-br from-purple-50 to-blue-50 p-6">
        <div className="max-w-2xl mx-auto">
          <div className="bg-white rounded-2xl shadow-xl p-8">
            <div className="text-center mb-8">
              <Heart className="w-16 h-16 mx-auto mb-4 text-purple-600" />
              <h1 className="text-3xl font-bold text-gray-800 mb-2">Your Wellness Report</h1>
            </div>

            <div className={`${details.bgColor} ${details.borderColor} border-2 rounded-xl p-6 mb-6`}>
              <div className="text-center mb-4">
                <div className={`text-5xl font-bold ${details.color} mb-2`}>{score}/18</div>
                <div className={`text-2xl font-semibold ${details.color}`}>{details.level}</div>
              </div>
              <p className="text-gray-700 text-center">{details.message}</p>
            </div>

            {details.tips.length > 0 && (
              <div className="bg-gray-50 rounded-xl p-6 mb-6">
                <h3 className="font-semibold text-gray-800 mb-3 flex items-center">
                  <CheckCircle className="w-5 h-5 mr-2 text-purple-600" />
                  Tips for Improvement
                </h3>
                <ul className="space-y-2">
                  {details.tips.map((tip, index) => (
                    <li key={index} className="text-gray-700 flex items-start">
                      <span className="text-purple-600 mr-2">â€¢</span>
                      {tip}
                    </li>
                  ))}
                </ul>
              </div>
            )}

            <div className="grid grid-cols-2 gap-4">
              <button
                onClick={() => setShowHistory(true)}
                className="bg-gray-100 hover:bg-gray-200 text-gray-800 font-semibold py-3 px-6 rounded-lg transition"
              >
                View History
              </button>
              <button
                onClick={handleReset}
                className="bg-purple-600 hover:bg-purple-700 text-white font-semibold py-3 px-6 rounded-lg transition"
              >
                Check Again
              </button>
            </div>
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-purple-50 to-blue-50 p-6">
      <div className="max-w-2xl mx-auto">
        <div className="bg-white rounded-2xl shadow-xl p-8">
          <div className="flex justify-between items-start mb-8">
            <div className="flex-1 text-center">
              <Heart className="w-16 h-16 mx-auto mb-4 text-purple-600" />
              <h1 className="text-3xl font-bold text-gray-800 mb-2">Daily Wellness Checker</h1>
              <p className="text-gray-600">Check how healthy your daily routine is</p>
            </div>
            {history.length > 0 && (
              <button
                onClick={() => setShowHistory(true)}
                className="bg-purple-100 hover:bg-purple-200 text-purple-700 px-4 py-2 rounded-lg transition flex items-center"
              >
                <Calendar className="w-4 h-4 mr-2" />
                History
              </button>
            )}
          </div>

          <div className="space-y-6">
            {questions.map((q) => {
              const Icon = q.icon;
              return (
                <div key={q.id} className="bg-gray-50 rounded-xl p-6">
                  <div className="flex items-center mb-4">
                    <Icon className="w-6 h-6 mr-3 text-purple-600" />
                    <h3 className="font-semibold text-gray-800">{q.question}</h3>
                  </div>
                  <div className="space-y-2">
                    {q.options.map((option) => (
                      <label
                        key={option.value}
                        className={`flex items-center p-3 rounded-lg cursor-pointer transition ${
                          answers[q.id] === option.value
                            ? 'bg-purple-100 border-2 border-purple-600'
                            : 'bg-white border-2 border-gray-200 hover:border-purple-300'
                        }`}
                      >
                        <input
                          type="radio"
                          name={q.id}
                          value={option.value}
                          checked={answers[q.id] === option.value}
                          onChange={() => handleAnswer(q.id, option.value)}
                          className="mr-3"
                        />
                        <span className="text-gray-700">{option.label}</span>
                      </label>
                    ))}
                  </div>
                </div>
              );
            })}
          </div>

          <button
            onClick={handleSubmit}
            className="w-full mt-8 bg-purple-600 hover:bg-purple-700 text-white font-semibold py-4 px-6 rounded-lg transition shadow-lg"
          >
            Check My Wellness Score
          </button>
        </div>
      </div>
    </div>
  );
}
